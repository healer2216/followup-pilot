### 改进 2：可视化多 Agent 工作流

#### 前端修改（待实施）

在加载提示中显示实时进度：

```javascript
async function generatePlan() {
  const summaryText = document.getElementById('summary-input').value.trim();
  if (!summaryText) {
    alert('请输入出院小结文本');
    return;
  }
  
  // 显示初始加载状态
  showAgentWorkflow([
    { agent: '分诊官', status: 'pending', icon: 'stethoscope' },
    { agent: 'RAG检索', status: 'pending', icon: 'search' },
    { agent: '规划师', status: 'pending', icon: 'calendar' },
    { agent: '审核员', status: 'pending', icon: 'shield-check' },
  ]);
  
  try {
    const res = await apiCall('/api/plan', { 
      summary_text: summaryText,
      session_id: 'web_' + Date.now()
    });
    
    if (!res.success) { 
      alert(res.error || '生成失败'); 
      return; 
    }
    
    // 更新工作流状态
    updateAgentWorkflow(res.data.agent_logs);
    
    planData = res.data;
    renderPlan(planData);
    switchPage('plan');
  } catch (e) {
    alert('请求失败: ' + e.message);
  } finally {
    hideLoading();
  }
}

function showAgentWorkflow(agents) {
  const overlay = document.getElementById('loading-overlay');
  const box = overlay.querySelector('.loading-box') || createLoadingBox();
  
  box.innerHTML = `
    <div class="loading-spinner"></div>
    <div class="loading-text">多 Agent 正在协作...</div>
    <div class="agent-workflow mt-4 space-y-2">
      ${agents.map(a => `
        <div class="agent-step flex items-center gap-2 text-sm" data-agent="${a.agent}">
          <i data-lucide="${a.icon}" class="w-4 h-4 text-gray-400"></i>
          <span>${a.agent}</span>
          <span class="ml-auto text-xs text-gray-400 status-badge">等待中...</span>
        </div>
      `).join('')}
    </div>
  `;
  
  overlay.classList.add('active');
  setTimeout(safeCreateIcons, 50);
}

function updateAgentWorkflow(logs) {
  logs.forEach(log => {
    const step = document.querySelector(`[data-agent="${log.agent.split(' ')[0]}"]`);
    if (step) {
      const badge = step.querySelector('.status-badge');
      badge.textContent = `✅ 完成 (${log.duration}s)`;
      badge.className = 'ml-auto text-xs text-green-600 status-badge';
    }
  });
}
```

      <i data-lucide="check-circle-2" class="w-5 h-5 text-green-500"></i>

      </span>

    <!-- 解析质量指标 -->
